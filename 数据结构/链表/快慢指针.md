### 快慢指针

#### 判断链表中是否有环

```C++
class Solution 
{
public:
    bool hasCycle(ListNode *head) 
    {
        if(!head || !head->next)
        {
            return false;
        }
        ListNode* slow = head;
        ListNode* fast = head->next;
        
        while(fast && fast->next)
        {
            slow = slow->next;
            fast = fast->next->next;
            if(slow == fast)
            {
                return true;
            }
        }
        return false;
    }
};
```
